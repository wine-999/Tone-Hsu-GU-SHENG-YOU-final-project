import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import matplotlib.dates as mdates # Import mdates

# 確保 'Date' 欄位是日期時間格式 (如果之前沒有執行過轉換)
df['Date'] = pd.to_datetime(df['Date'], format='%d/%m/%Y')

# 提取年份和月份，並按年份和月份分組，計算總租借單車數量 (用於折線圖)
df['Year'] = df['Date'].dt.year
df['Month'] = df['Date'].dt.month
monthly_rentals_sum = df.groupby(['Year', 'Month'])['Rented Bike Count'].sum().reset_index()

# 合併年份和月份為一個日期欄位，方便繪圖並按日期排序
monthly_rentals_sum['YearMonth'] = pd.to_datetime(monthly_rentals_sum['Year'].astype(str) + '-' + monthly_rentals_sum['Month'].astype(str) + '-01')
monthly_rentals_sum = monthly_rentals_sum.sort_values(by='YearMonth')


# 創建圖表和主 Y 軸
fig, ax1 = plt.subplots(figsize=(12, 6))

# 繪製長條圖 (下層) 使用 Matplotlib bar - 顯示每天總數
# 將長條圖繪製在主 Y 軸 (ax1) 上
ax1.bar(df['Date'], df['Rented Bike Count'], color='blue', label='Daily Total (Bars)')
ax1.set_ylabel('Daily Rented Bike Count', color='blue')
ax1.tick_params(axis='y', labelcolor='blue')


# 創建第二個 Y 軸，與主 Y 軸共享 X 軸
ax2 = ax1.twinx()

# 繪製折線圖 (上層) 使用 Matplotlib plot - 顯示每月總數
# 將折線圖繪製在第二個 Y 軸 (ax2) 上
ax2.plot(monthly_rentals_sum['YearMonth'], monthly_rentals_sum['Rented Bike Count'], color='red', marker='o', label='Monthly Total (Line)')
ax2.set_ylabel('Monthly Rented Bike Count', color='red')
ax2.tick_params(axis='y', labelcolor='red')


# 設定 X 軸為日期格式 (使用 ax1 來設定 X 軸)
ax1.xaxis.set_major_locator(mdates.MonthLocator(interval=1)) # 每月一個主要刻度
ax1.xaxis.set_major_formatter(mdates.DateFormatter('%Y-%m')) # 設定日期格式

# 設定標題和 X 軸標籤
plt.title('Daily Total (Bars) and Monthly Total (Line) Rented Bike Count')
ax1.set_xlabel('Date') # X 軸標籤只需要設定一次

# 旋轉 X 軸標籤
plt.xticks(rotation=45, ha='right') # ha='right' 調整對齊方式

# 顯示圖例 (需要合併兩個軸的圖例)
lines1, labels1 = ax1.get_legend_handles_labels()
lines2, labels2 = ax2.get_legend_handles_labels()
ax2.legend(lines1 + lines2, labels1 + labels2, loc='upper left')


plt.tight_layout() # 自動調整佈局
plt.show()
